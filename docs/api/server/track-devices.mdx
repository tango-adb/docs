# `trackDevices`

```ts
import { AddEventListener } from "@yume-chan/event";
import { DeviceObserver } from "@yume-chan/adb";
import { AdbDaemonWebUsbDevice } from "@yume-chan/adb-daemon-webusb";

export declare interface AdbServerDeviceObserver
  extends DeviceObserver<AdbServerClient.Device> {
  onError: AddEventListener<Error>;
  onDeviceAdd: AddEventListener<AdbServerClient.Device[], unknown>;
  onDeviceRemove: AddEventListener<AdbServerClient.Device[], unknown>;
  onListChange: AddEventListener<AdbServerClient.Device[], unknown>;
  current: AdbServerClient.Device[];

  stop(): Promise<void>;
}

declare class AdbServerClient {
  trackDevices(): Promise<AdbServerDeviceObserver>;
}
```

Get notified when a device is connected or disconnected.

:::note

See [Watch devices](/tango/server/watch.mdx) page for details.

:::

:::info[Equivalent ADB Command]

There is no equivalent ADB command.

:::
